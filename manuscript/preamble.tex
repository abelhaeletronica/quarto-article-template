\newcommand{\articletitle}[2]{%
  \begin{center}
    {\mdseries\fontsize{19}{19}\selectfont\sffamily #1\par}
    \vspace{0.3em}
    {\Large\sffamily #2}
  \end{center}
}

\newenvironment{articleabstract}{%
  \begin{addmargin}[1.5em]{1.5em}
  \small\setlength{\parindent}{0pt}
}{%
  \end{addmargin}
  \vspace{0.5em}
}

\newcommand{\authorname}[1]{%
  \begin{center}
    {\sffamily\large\bfseries #1}
  \end{center}
}

\newcommand{\authoraffil}[1]{%
  \begin{center}
    {\footnotesize\sffamily
      \renewcommand{\arraystretch}{0.85}%
      \begin{tabular}{@{}c@{}}
        #1
      \end{tabular}}
  \end{center}
}
